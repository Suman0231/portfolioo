8. Check if given number is a perfect square.

#include <stdio.h> 

int main() {
    int num, i;
    int check = 0;

    printf("Enter a number: ");
    scanf("%d", &num);

    for(i = 1; i * i <= num; i++) {
        if(i * i == num) {
            check = 1;
            break;
        }
    }

    if(check==1)
        printf("%d is a Perfect Square\n", num);
    else
        printf("%d is NOT a Perfect Square\n", num);

    return 0;
}
