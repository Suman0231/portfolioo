String Handling Functions in C

1. strlen()

Function: Returns total number of characters in a given string.

#include <stdio.h>
#include <string.h>

int main()
{
    char a[50];
    int l;

    printf("Enter any string: ");
    scanf("%s", a);

    l = strlen(a);
    printf("The length is %d\n", l);

    return 0;
}

2. strrev()

#include <stdio.h>
#include <string.h>

int main()
{

5. strcpy()

Function: Copies one string to another.

}

6. strcat()

Function: Concatenates two strings.

#include <stdio.h>
#include <string.h>

int main()
{
    char a[50], b[50];

    printf("Enter first string: ");
    scanf("%s", a);

    printf("Enter second string: ");
    scanf("%s", b);

    strcat(a, b);
    printf("The added string is %s\n", a);

    return 0;
}

7. strcmp()

Function: Compares two strings.

#include <stdio.h>
#include <string.h>

int main()
{
    char a[10]="ram", b[10]="amrit",c[10]="sahil",d[10]="ram";
    int x;

    x = strcmp(a, b);
    printf("The compared value is %d\n", x);
    x = strcmp(a, c);
    printf("The compared value is %d\n", x);

}

8. Palindrome Check

Function: Checks whether a given string is a palindrome.

#include <stdio.h>
#include <string.h>

int main()
{
    char a[50], b[50];

    printf("Enter any string: ");
    scanf("%s", a);

    strcpy(b, a);
    strrev(b);

    if(strcmp(a, b) == 0)
        printf("The string is a palindrome\n");
    else
        printf("The string is not a palindrome\n");

    return 0;
}
