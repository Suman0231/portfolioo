2. JavaScript Arithmetic Operations Example

<!DOCTYPE html>
<html lang="en">
<head>
    

    <title>Arithmetic Operators Example</title>
</head>
<body>
    <h2>Arithmetic Operators in JavaScript</h2>

    <script>
        var a = 10;
        var b = 3;

        document.write("a = " + a + "<br>");
        document.write("b = " + b + "<br><br>");

        // Addition
        document.write("Addition (a + b) = " + (a + b) + "<br>");

        // Subtraction
        document.write("Subtraction (a - b) = " + (a - b) + "<br>");

        // Multiplication
        document.write("Multiplication (a * b) = " + (a * b) + "<br>");

        // Division
        document.write("Division (a / b) = " + (a / b) + "<br>");

        // Modulus (a % b)
        document.write("Modulus (a % b) = " + (a % b) + "<br>");

        
        // Increment
        var x = a;
        x++;
        document.write("Increment (a++) = " + x + "<br>");

        // Decrement
        var y = b;
        y--;
        document.write("Decrement (b--) = " + y + "<br>");
    </script>

</body>
</html>

    <title>Greatest of Two Numbers</title>

    <h2>Find the Greatest of Two Numbers</h2>

    <script>
        // Taking input using prompt (inputs are strings)
        var num1 = prompt("Enter the first number:");
        var num2 = prompt("Enter the second number:");

        // Convert to numbers
        num1 = Number(num1);
        num2 = Number(num2);

        // Check which is greater
        if (num1 > num2) {
            document.write("The greatest number is: " + num1);
        } 
        else if (num2 > num1) {
            document.write("The greatest number is: " + num2);
        } 
        else {
            document.write("Both numbers are equal.");
        }
    </script>

</body>

5. Difference between post and pre increments. 

<!DOCTYPE html>
<html lang="en">
<head>
    

    <title>Pre & Post Increment/Decrement</title>
</head>
<body>
    <h2>Pre and Post Increment/Decrement Example</h2>

    <script>
        var a = 5;
        var b = 5;

        document.write("<b>Initial Values:</b><br>");
        document.write("a = " + a + "<br>");
        document.write("b = " + b + "<br><br>");

        // POST-INCREMENT (a++)
        var postInc = a++;
        document.write("<b>Post-Increment (a++)</b><br>");
        document.write("Returned value = " + postInc + "<br>");
        document.write("New value of a = " + a + "<br><br>");

        // PRE-INCREMENT (++b)
        var preInc = ++b;
        document.write("<b>Pre-Increment (++b)</b><br>");
        document.write("Returned value = " + preInc + "<br>");
        document.write("New value of b = " + b + "<br><br>");

        var c = 5;
        var d = 5;

        document.write("<b>Initial Values:</b><br>");
        document.write("c = " + c + "<br>");
        document.write("d = " + d + "<br><br>");

        // POST-DECREMENT (c--)
        var postDec = c--;
        document.write("<b>Post-Decrement (c--)</b><br>");
        document.write("Returned value = " + postDec + "<br>");
        document.write("New value of c = " + c + "<br><br>");

        // PRE-DECREMENT (--d)
        var preDec = --d;
        document.write("<b>Pre-Decrement (--d)</b><br>");
        document.write("Returned value = " + preDec + "<br>");
        document.write("New value of d = " + d + "<br>");
    </script>

</body>
</html>
