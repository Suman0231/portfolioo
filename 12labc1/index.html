2. WAP to find factorial of a number using recursion

#include <stdio.h>

int factorial(int n) {
    if(n == 0) 
        return 1;
    return n * factorial(n - 1);
}

int main() {
    int num;
    printf("Enter a number: ");
    scanf("%d", &num);

    printf("Factorial = %d", factorial(num));
    return 0;
}

3. WAP illustrating Call by Value and Call by Reference

#include <stdio.h>

void callByValue(int a) {
    a = a + 10;
    printf("Inside callByValue: %d\n", a);
}

void callByReference(int *b) {
    *b = *b + 10;
    printf("Inside callByReference: %d\n", *b);
}

int main() {
    int x = 5, y = 5;

    printf("Before callByValue: %d\n", x);
    callByValue(x);
    printf("After callByValue: %d\n\n", x);

    printf("Before callByReference: %d\n", y);
    callByReference(&y);
    printf("After callByReference: %d\n", y);

    return 0;
}

5. WAP in C to input roll, name and marks of n students and display them

#include <stdio.h>

struct Student {
    int roll;
    char name[30];
    float marks;
};

int main() {
    int n;
    printf("Enter number of students: ");
    scanf("%d", &n);

    struct Student s[n];

    // Input details
    for(int i = 0; i < n; i++) {
        printf("\nEnter details of student %d\n", i + 1);

        printf("Enter Roll Number: ");
        scanf("%d", &s[i].roll);

        printf("Enter Name: ");
        scanf("%s", s[i].name);

        printf("Enter Marks: ");
        scanf("%f", &s[i].marks);
    }

    // Display details
    printf("\n------ Student Details ------\n");
    for(int i = 0; i < n; i++) {
        printf("\nStudent %d:\n", i + 1);
        printf("Roll  : %d\n", s[i].roll);
        printf("Name  : %s\n", s[i].name);
        printf("Marks : %.2f\n", s[i].marks);
    }

    return 0;
}

6. WAP showing illustration og Nested Structure

#include <stdio.h>

struct Address {
    char city[20];
    int pincode;
};

struct Student {
    int roll;
    char name[20];
    struct Address addr;   // Nested structure
};

int main() {
    struct Student s;

    printf("Enter roll number: ");
    scanf("%d", &s.roll);

    printf("Enter name: ");
    scanf("%s", s.name);

    printf("Enter city: ");
    scanf("%s", s.addr.city);

    printf("Enter pincode: ");
    scanf("%d", &s.addr.pincode);

    printf("\n--- Student Details ---\n");
    printf("Roll No: %d\n", s.roll);
    printf("Name: %s\n", s.name);
    printf("City: %s\n", s.addr.city);
    printf("Pincode: %d\n", s.addr.pincode);

    return 0;
}

6. WAP in C to input roll, name and marks of student using Union

#include <stdio.h>

union Student {
    int roll;
    char name[30];
    float marks;
};

int main() {
    union Student s;

    printf("Enter Roll Number: ");
    scanf("%d", &s.roll);
    printf("Roll = %d\n\n", s.roll);

    printf("Enter Name: ");
    scanf("%s", s.name);
    printf("Name = %s\n\n", s.name);

    printf("Enter Marks: ");
    scanf("%f", &s.marks);
    printf("Marks = %.2f\n\n", s.marks);

    printf("NOTE:\n");
    printf("Name = %s\n\n", s.name);
    printf("Union stores only one value at a time.\n");
    printf("So only the LAST entered value is actually stored.\n");

    return 0;
}

7. WAP in C using sizeof() to illustrate size difference between Structure and Union

#include <stdio.h>

struct S {
    int roll;      // 4 bytes
    float marks;   // 4 bytes
    char grade;    // 1 byte 
};

union U {
    int roll;      // 4 bytes
    float marks;   // 4 bytes
    char grade;    // 1 byte
};

int main() {
    struct S student1;
    union U student2;

    printf("Size of Structure = %lu bytes\n", sizeof(student1)); //%lu format specifier is used to print unsigned long integers
    printf("Size of Union     = %lu bytes\n", sizeof(student2));

    printf("\nNOTE:\n");
    printf("Structure size = sum of all members (plus padding).\n");
    printf("Union size = size of the largest member.\n");

    return 0;
}
